import ShowResourcePage, { ShowPageItem } from '@/Components/ShowPage/ShowResourcePage'
import { useMemo, useState } from 'react'
import DeleteModal from '@/ui/Modal/DeleteModal'
import { #[ModelName] } from '@/interfaces/data_interfaces'

interface Props {
  #[modelName]: #[ModelName]
}

export default function MetaGroupShow({ #[modelName] }: Readonly<Props>) {

  const [showDeleteModal, setShowDeleteModal] = useState(false)

  const displayedValues = useMemo(() => {
    return [
      {
         id: 1,
         label: 'Name',
         content: '',
         type: 'text',
      },
    ] as ShowPageItem[]
  }, [])

  return (
    <ShowResourcePage
      title={''}
      items={displayedValues}
      backUrl={route('#[url].index')}
      editUrl={route('#[url].edit', #[modelName].id)}
      onDeleteClick={() => {
        setShowDeleteModal(true)
      }}
    >
     {/**more content**/}
     {showDeleteModal && (
        <DeleteModal
           setShowModal={setShowDeleteModal}
           title={`Delete Record`}
           url={route('#[url].destroy', #[modelName].id)}
        >
           <p>Are you sure you want to delete record?</p>
        </DeleteModal>
     )}
    </ShowResourcePage>
  )
}
